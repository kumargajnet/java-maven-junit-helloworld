pipeline {
    agent any

    tools {
        maven "maven"
    }

    environment {
        // ---- Your Real Servers ----
        TOMCAT_SERVER = "65.2.172.200"
        TOMCAT_USER   = "ubuntu"
        SSH_KEY_PATH  = "/var/lib/jenkins/.ssh/jenkins_key"

        NEXUS_URL     = "35.154.120.194:8081"
        NEXUS_REPO    = "maven-releases"
        NEXUS_CREDS   = "Nexus-credentials"
    }

    stages {

        stage('Build WAR') {
            steps {
                echo "Building WAR..."
                sh 'mvn clean package -DskipTests'
            }
        }

        stage('Extract Version') {
            steps {
                script {
                    env.VERSION = sh(
                        script: "mvn help:evaluate -Dexpression=project.version -q -DforceStdout",
                        returnStdout: true
                    ).trim()
                    echo "Project Version = ${VERSION}"
                }
            }
        }

        stage('Upload to Nexus') {
            steps {
                script {
                    def war = sh(script: 'find target -name "*.war" -print -quit', returnStdout: true).trim()

                    echo "Uploading WAR to Nexus: ${war}"

                    nexusArtifactUploader(
                        nexusVersion: 'nexus3',
                        protocol: 'http',
                        nexusUrl: "${NEXUS_URL}",
                        groupId: 'com.example.helloworld',
                        version: "${VERSION}",
                        repository: "${NEXUS_REPO}",
                        credentialsId: "${NEXUS_CREDS}",
                        artifacts: [[
                            artifactId: 'helloworld-app',
                            classifier: '',
                            file: war,
                            type: 'war'
                        ]]
                    )
                }
            }
        }

        stage('Deploy to Tomcat') {
            steps {
                script {

                    def warFile = sh(script: 'find target -name "*.war" -print -quit', returnStdout: true).trim()

                    echo "Deploying WAR to Tomcat..."

                    sh """
                        scp -i ${SSH_KEY_PATH} -o StrictHostKeyChecking=no ${warFile} ${TOMCAT_USER}@${TOMCAT_SERVER}:/tmp/

                        ssh -i ${SSH_KEY_PATH} -o StrictHostKeyChecking=no ${TOMCAT_USER}@${TOMCAT_SERVER} '
                            sudo rm -rf /opt/tomcat/webapps/helloworld*
                            sudo mv /tmp/*.war /opt/tomcat/webapps/helloworld.war
                            sudo systemctl restart tomcat
                        '
                    """
                }
            }
        }

        stage('Show URLs') {
            steps {
                echo "APPLICATION URL:"
                echo "http://${TOMCAT_SERVER}:8080/helloworld/"

                echo "NEXUS ARTIFACT URL:"
                echo "http://${NEXUS_URL}/repository/${NEXUS_REPO}/com/example/helloworld/helloworld-app/${VERSION}/helloworld-app-${VERSION}.war"
            }
        }
    }

    post {
        success {
            echo "Pipeline completed successfully!"
        }
        failure {
            echo "Pipeline failed!"
        }
    }
}
